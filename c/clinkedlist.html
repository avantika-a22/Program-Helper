C Program to understand Linked list.

#include<conio.h>

#include<stdio.h>

#include<malloc.h>

struct node

{

  int item;

  node *next;

} *start;

// Function to create linked list

void create(int data)

{

  struct node *p,*tmp;

  p = (struct node *)malloc(sizeof(struct node));

  p->item=data;

  p->next=start;

  start = p;

}

// Function to add data to beginning of list

void addbeginning(int data)

{

  struct node *p;

  p=(struct node *)malloc(sizeof(struct node));

  p -> item = data;

  p -> next = start;

  start = p;

}

// Function to add data to last of list

void addLast(int data)

{

  struct node *p,*tmp;

  p=(struct node *)malloc(sizeof(struct node));

  tmp=start;

  p->item=data;

  while(tmp->next!=NULL)

  {

    tmp = tmp -> next;

  }

  tmp -> next = p;

  p -> next = NULL;

}

// Function to add data in between list

void addmiddle(int data, int after)

{

  struct node *p,*tmp;

  p=(struct node * )malloc(sizeof(struct node));

  tmp=start;

  p->item=after;

  while(tmp->item!=data)

  {

    tmp = tmp -> next;

  }

  p -> next = tmp -> next;

  tmp -> next = p;

  }

void deletebeginning()

{

  struct node *p;

  p = start;

  start = start -> next;

  free(p);    // to free the memory

}

// Function to delete data from last

void deletelast()

{

  struct node *p;

  p = start;

  while(p -> next -> next != NULL)

  {

   p = p -> next;

  }

p -> next = NULL;

}

// Function to delete in between list

void deleteafter(int data)

{

      struct node *p;

       p = start;

      while(p -> item != data)

      {

       p = p -> next;

      }

      p -> next = p -> next -> next;

}

// Function to search element in linked list

void search(int data)

{

  int count = 0;

  struct node *p;

  p = start;

  while(p != NULL)

  {

    if(p -> item == data)

  {

    printf("Element found at %d position",count+1);

  return;

}

p = p -> next;

count ++;

}

if (p == NULL)

{

  printf("Data not found");

}

}

// Function to reverse linked list

void reverse()

{

  struct node *p1,*p2;

  p1 = NULL;

  p2 = NULL;

  while(start != NULL){

      p1 = start -> next;

      start -> next = p2;

      p2 = start;

      start = p1;

}

start = p2;

while(start != NULL)

{

  printf("\n%d",start -> item);

  start = start -> next;

}

}

int i;

struct node *tmp;

tmp = start;

while(tmp -> next != NULL)
