File Handling in C++.

1. To open a file and retrieve data from file.

#include<iostream.h>

#include<fstream.h>

#include<conio.h>

void main()

{

char a[10];

// ifstream is input stream and fin is object of class ifstream

ifstream fin;

fin.open("file.dat");

fin>>a;

cout<<a;

getch();

}

Output

hello

2. To open a file and copy data from program variable to file.

#include<iostream.h>

#include<fstream.h>

#include<conio.h>

void main()

{

char a[6] = 'hello';

// ofstream is output stream class and fout is object of ofstream class

ofstream fout;

fout.open("file.dat"); // file.dat file is created if not already created in the same folder.

fout<<a;

getch();

}

Output

'hello' is copied to file.dat
