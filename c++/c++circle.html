- C++ Program which accepts the radius of a circle from the user and compute the area and circumference

#include<iostream>

using namespace std;

class circle

{

private:

  float r;

public:

  circle(float n)

  {

    r = n;

  }

  display()

  {

    int area,cir;

    area = 3.14 * r * r;

    cir = 2 * 3.14 * r;

    cout<<"Area = "<<area<<endl;

    cout<<"Circumference = "<<cir<<endl;

  }

};

int main()

{

circle obj(10);

obj.display();

return 0;

}

Output

Area : 314

Circumference : 62.8
