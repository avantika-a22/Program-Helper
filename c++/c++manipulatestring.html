C++ Program for Selection Sort

#include<iostream.h>

#include<conio.h>

#define Size 10

void main()

{

int a[Size];

int i, j, temp;

cout<<"Enter elements of the array: ";

for(i = 0;i < Size; i++)

{

cin(" %d",&a[i]);

}

for(i = 0;i < Size - 1; i++)

{

for(j = i + 1;j < Size; j++)

{

if(a[i] > a[j])

{

temp=a[i];

a[i]=a[j];

a[j]=temp;

}

}

}

cout<<"Sorted array is : \n";

for(i = 0;i < Size; i++){

cout("%d",a[i]);

}

getch();

}
