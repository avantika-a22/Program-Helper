Copy Constructor

#include<iostream>

using namespace std;

Class Copy

{

private:

int a , b;

public:

Copy(int m, int n)

{ a = m; b = n; }

// Copy constructor

Copy(const Copy &p2)

{a = p2.a; b = p2.b; }

int getA() { return a; }

int getB() { return b; }

};

int main()

{

Copy p1(5, 15); // Normal constructor

Copy p2 = p1; // Copy constructor

cout<< "p1.a = " << p1.getA() << ", p1.b = "<< p1.getB();

cout << "\np2.a = " << p2.getA()<< ", p2.b = " <<p2.getB();

return 0;

}

Output

p1.a = 5, p1.b = 15

p2.a = 5, p2.b = 15

Multiple Constructors

#include <iostream>

using namespace std;

class multiple

{

public:

float sum;

//constructor with two parameters

multiple(int a , int b)

{

sum = a + b;

}

//constructor with no parameters

multiple()

{

sum = 0;

}

void disp()

{

cout<< sum<< endl;

}

};

int main()

{

multiple o(20 , 40);

multiple o2;

o.disp();

o2.disp();

return 1;

}

Output

60

0

Destructors

Destructor is a member function which destructs or deletes an object.

There can only be one destructor in a class.

#include<iostream>

using namespace std;

class destructor

{

public:

//constructor

destructor()

{

cout<<"It is constructor"<<endl;

}

//destructor

~destructor()

{

cout<<"It is destructor"<<endl;

}

void disp()

{

cout<<"Hello Destructor"<<endl;

}

};

int main()

{

destructor obj;

obj.disp();

return 0;

}

Output

It is constructor

Hello Destructor

It is destructor
